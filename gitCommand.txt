git branch -M main
git remote add origin https://github.com/onin-hub/sample_html.git

Tip to get only the remote URL:
git config --get remote.origin.url

remote other url:
git remote set-url origin <repo_link>

clone the specfic branch
git clone --branch <branchname> <repo_link>

// delete branch locally
git branch -d localBranchName

// delete branch remotely
git push origin --delete remoteBranchName

//merge
git merge <branch name you want to merge to>
